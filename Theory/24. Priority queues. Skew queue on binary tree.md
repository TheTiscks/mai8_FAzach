## Приоритетные очереди. Косая приоритетная очередь, её реализация на базе структуры данных вида бинарное дерево. Операции вставки элемента, поиска минимума/максимума, удаления минимума/максимума, слияния с разрушением и без разрушения исходных косых приоритетных очередей; их вычислительные сложности и реализация на языке C.

Косая куча — это саморегулирующаяся бинарная куча (обычно min-heap), в которой слияние является ключевой операцией.

В отличие от leftist heap, здесь нет дополнительного поля npl.

Саморегуляция достигается за счёт безусловной перестановки левого и правого ребёнка при каждом слиянии.

Это даёт амортизированную сложность O(log n) для всех операций (не гарантированную, как в leftist, но очень хорошую на практике).

•Обладает такими же свойствами, как и левосторонняя куча, но имеет отличие в процессе слияния двух куч.

````C
typedef struct Node {
    int key;                   // значение (приоритет)
    struct Node *left;
    struct Node *right;
} Node;

typedef Node* SkewHeap;        // очередь — указатель на корень
````

Операции и вычислительные сложности

* Поиск минимума/максимума — O(1) (корень кучи)
* Вставка элемента — O(log n) амортизированная (создать узел + слияние)
* Удаление минимума/максимума — O(log n) амортизированная (удалить корень + слияние поддеревьев)
* Слияние двух куч — O(log n) амортизированная. Схоже с алгоритмом для левосторонней кучи, но перестановка левого и правого поддерева БЕЗУСЛОВНА
  
* (
1. Если одна из куч пуста — возвращаем другую.
2. Сравниваем корни двух куч. Если корень второй кучи меньше — меняем кучи местами (чтобы корень результата имел минимальный ключ).
3. Корнем результата становится корень первой кучи.
4. Безусловно меняем местами левое и правое поддеревья этого корня (это ключевое действие).
5. Рекурсивно сливаем вторую кучу с тем поддеревом, которое теперь стало правым (после перестановки).
6. Присваиваем результат рекурсии правому ребёнку корня.
7. Возвращаем новый корень.
  )

Реализация на Си

````C
#include <stdlib.h>

// Ключевая операция: слияние с разрушением (амортизированная O(log n))
SkewHeap merge(SkewHeap h1, SkewHeap h2) {
    if (!h1) return h2;
    if (!h2) return h1;

    // Обеспечиваем h1->key ≤ h2->key
    if (h1->key > h2->key) {
        SkewHeap temp = h1;
        h1 = h2;
        h2 = temp;
    }

    // Безусловная перестановка детей
    SkewHeap temp = h1->left;
    h1->left = h1->right;
    h1->right = temp;

    // Рекурсивно сливаем бывшее правое поддерево с h2
    h1->right = merge(h1->right, h2);

    return h1;
}

// Поиск минимума (min-heap)
int find_min(SkewHeap h) {
    if (!h) {
        // ошибка: пустая куча
        exit(1);
    }
    return h->key;
}

// Вставка
SkewHeap insert(SkewHeap h, int key) {
    SkewHeap new_node = malloc(sizeof(Node));
    new_node->key = key;
    new_node->left = new_node->right = NULL;
    return merge(h, new_node);
}

// Удаление минимума
int delete_min(SkewHeap *h) {
    if (!*h) {
        // ошибка
        exit(1);
    }
    int min_val = (*h)->key;
    SkewHeap left = (*h)->left;
    SkewHeap right = (*h)->right;
    free(*h);
    *h = merge(left, right);
    return min_val;
}

// Слияние без разрушения исходных (копируем путь слияния)
SkewHeap merge_non_destructive(SkewHeap h1, SkewHeap h2) {
    if (!h1) return h2 ? h2 : NULL;  // простая копия, если один пуст
    if (!h2) return h1;

    SkewHeap root;
    if (h1->key <= h2->key) {
        root = malloc(sizeof(Node));
        root->key = h1->key;
        root->left = h1->left;
        root->right = merge_non_destructive(h1->right, h2);
    } else {
        root = malloc(sizeof(Node));
        root->key = h2->key;
        root->left = h2->left;
        root->right = merge_non_destructive(h1, h2->right);
    }

    // Безусловная перестановка
    SkewHeap temp = root->left;
    root->left = root->right;
    root->right = temp;

    return root;
}

// Освобождение кучи
void free_skew_heap(SkewHeap h) {
    if (h) {
        free_skew_heap(h->left);
        free_skew_heap(h->right);
        free(h);
    }
}
````
